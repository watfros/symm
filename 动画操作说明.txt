
# 选中所有的原子
set allatom [atomselect top all]
# 获取分子质心,变化为负向量，转为矩阵，并设为m0
set m0 [transoffset [vecinvert [measure center $allatom weight mass]]]
# 施加平移操作
$allatom move $m0

# 每隔100毫秒，对每个原子施加移动操作，形成动画。

array set vatom_ ""

set vscale_($i) [vecdot $vec $vatom_($i)]
